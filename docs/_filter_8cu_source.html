63  if (sinogram.valid()) {

87 auto Filter::processor(const int deviceID) -> void {
88  //nvtxNameOsThreadA(pthread_self(), "Filter");
89  CHECK(cudaSetDevice(deviceID));
90  auto sinoFreq = ddrf::cuda::make_device_ptr<cufftComplex,
91  ddrf::cuda::async_copy_policy>(
92  numberOfProjections_ * ((numberOfDetectors_ / 2.0) + 1));
93  dim3 dimBlock(blockSize2D_, blockSize2D_);
94  dim3 dimGrid((int) ceil((numberOfDetectors_ / 2.0 + 1) / (float) blockSize2D_),
95  (int) ceil(numberOfProjections_ / (float) blockSize2D_));
96  auto filterFunction_d = ddrf::cuda::make_device_ptr<float, ddrf::cuda::async_copy_policy>(filter_.size());
97  CHECK(cudaMemcpy(filterFunction_d.get(), filter_.data(), sizeof(float)*filter_.size(), cudaMemcpyHostToDevice));
98  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Filter: Running Thread for Device " << deviceID;
99  while (true) {
100  auto sinogram = sinograms_[deviceID].take();
101  if (!sinogram.valid())
102  break;
103  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Filter: Filtering sinogram with Index " << sinogram.index();
104 
105  //forward transformation
106  CHECK_CUFFT(
107  cufftExecR2C(plansFwd_[deviceID],
108  (cufftReal* ) sinogram.container().get(),
109  thrust::raw_pointer_cast(&(sinoFreq[0]))));
110 
111  //Filtering
112  applyFilter<<<dimGrid, dimBlock, 0, streams_[deviceID]>>>(
113  (numberOfDetectors_ / 2) + 1, numberOfProjections_, sinoFreq.get(), filterFunction_d.get());
114 
115  CHECK(cudaPeekAtLastError());
116 
117  //reverse transformation
118  CHECK_CUFFT(
119  cufftExecC2R(plansInv_[deviceID],
120  thrust::raw_pointer_cast(sinoFreq.get()),
121  (cufftReal* ) sinogram.container().get()));
122  //wait until work on device is finished
123  CHECK(cudaStreamSynchronize(streams_[deviceID]));
124  results_.push(std::move(sinogram));
125 
126  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Filter: Filtering sinogram with Index " << sinogram.index() << " finished.";
127  }
128 }

154 auto Filter::designFilter() -> void {
155  int filterSize = numberOfDetectors_/2 + 1;
156  filter_.reserve(filterSize);
157  filter_.push_back(0.0);
158  for(auto i = 1; i < filterSize; i++){
159  //actual w at frequency axis
160  const float w = 2 * M_PI * i / (float)numberOfDetectors_;
161  if(w > M_PI*cutoffFraction_){
162  filter_.push_back(0.0);
163  continue;
164  }
165  float filterValue = 2 * i / (float)numberOfDetectors_; //* hanning(w, (float)1.0);
166  if(filterType_ == detail::FilterType::hamming)
167  filterValue *= hamming(w, cutoffFraction_);
168  else if(filterType_ == detail::FilterType::hanning)
169  filterValue *= hanning(w, cutoffFraction_);
170  else if(filterType_ == detail::FilterType::sheppLogan)
171  filterValue *= sheppLogan(w, cutoffFraction_);
172  else if(filterType_ == detail::FilterType::cosine)
173  filterValue *= cosine(w, cutoffFraction_);
174  filter_.push_back(filterValue/(float)numberOfDetectors_);
175  }
176 }

178 auto Filter::readConfig(const std::string& configFile) -> bool {
179  ConfigReader configReader = ConfigReader(configFile.data());
180  std::string filterType;
181  if (configReader.lookupValue("numberOfParallelProjections", numberOfProjections_)
182  && configReader.lookupValue("numberOfParallelDetectors", numberOfDetectors_)
183  && configReader.lookupValue("numberOfPixels", numberOfPixels_)
184  && configReader.lookupValue("blockSize2D_filter", blockSize2D_)
185  && configReader.lookupValue("filterType", filterType)
186  && configReader.lookupValue("cutoffFraction", cutoffFraction_)){
187  if(filterType == "ramp")
188  filterType_ = detail::FilterType::ramp;
189  else if(filterType == "sheppLogan")
190  filterType_ = detail::FilterType::sheppLogan;
191  else if(filterType == "hamming")
192  filterType_ = detail::FilterType::hamming;
193  else if(filterType == "hanning")
194  filterType_ = detail::FilterType::hanning;
195  else if(filterType == "cosine")
196  filterType_ = detail::FilterType::cosine;
197  else{
198  BOOST_LOG_TRIVIAL(error) << "recoLib::cuda::Filter: Requested filter mode not supported. Using Ramp-Filter.";
199  filterType_ = detail::FilterType::ramp;
200  }
201  return EXIT_SUCCESS;
202  }
203  return EXIT_FAILURE;
204 }
