88 auto OfflineSaver::saveImage(ddrf::Image<manager_type> image,
89  std::string path) -> void {
90  auto img = ddrf::MemoryPool<manager_type>::instance()->requestMemory(
91  memoryPoolIndex_);
92  std::copy(image.container().get(),
93  image.container().get() + image.size(),
94  img.container().get());
95  img.setIdx(image.index());
96  img.setPlane(image.plane());
97  img.setStart(image.start());
98  auto latency = img.duration();
99  if(latency < minLatency_)
100  minLatency_ = latency;
101  else if(latency > maxLatency_)
102  maxLatency_ = latency;
103  outputBuffers_[image.plane()].push_back(std::move(img));
104  if(mode_ == detail::RecoMode::offline){
105  if(outputBuffers_[image.plane()].full()){
106  writeTiffSequence(image.plane());
107  }
108  }else if(mode_ == detail::RecoMode::online){
109  tmrs_[image.plane()].stop();
110  double diff = tmrs_[image.plane()].elapsed();
111  if(diff < 0.02)
112  return;
113  std::string path = outputPath_ + fileName_ + "_plane"
114  + std::to_string(image.plane()) + "_" + "0" + ".tif";
115  auto tif = std::unique_ptr<::TIFF, detail::TIFFDeleter> { TIFFOpen(
116  path.c_str(), "wb") };
117  if (tif == nullptr)
118  throw std::runtime_error { "recoLib::OfflineSaver: Could not open file "
119  + path + " for writing." };
120  writeToTiff(tif.get(), std::move(image));
121  tmrs_[image.plane()].start();
122  }

125 auto OfflineSaver::writeTiffSequence(const int planeID) -> void {
126  if(outputBuffers_[planeID].count() == 0)
127  return;
128  std::string path = outputPath_ + fileName_ + "_plane"
129  + std::to_string(planeID) + "_" + std::to_string(fileIndex_[planeID]) + "seq.tif";
130  auto tif = std::unique_ptr<::TIFF, detail::TIFFDeleter> { TIFFOpen(
131  path.c_str(), "w8") };
132  if (tif == nullptr)
133  throw std::runtime_error { "recoLib::OfflineSaver: Could not open file "
134  + path + " for writing." };
135 
136  for(auto i = 0u; i < outputBuffers_[planeID].count(); i++){
137  BOOST_LOG_TRIVIAL(debug) << "writing out index " << i;
138  writeToTiff(tif.get(), std::move(outputBuffers_[planeID].at(i)));
139  if(TIFFWriteDirectory(tif.get()) != 1)
140  throw std::runtime_error{"savers::TIFF: tiffio error while writing to " + path};
141  }
142  ++fileIndex_[planeID];
143  outputBuffers_[planeID].clear();
144 }

174 auto OfflineSaver::readConfig(const std::string& configFile) -> bool {
175  ConfigReader configReader = ConfigReader(configFile.data());
176  std::string mode;
177  if (configReader.lookupValue("numberOfPixels", numberOfPixels_)
178  && configReader.lookupValue("outputPath", outputPath_)
179  && configReader.lookupValue("outputFileName", fileName_)
180  && configReader.lookupValue("numberOfDataFrames", numberOfFrames_)
181  && configReader.lookupValue("numberOfPlanes", numberOfPlanes_)
182  && configReader.lookupValue("outputBufferSize", circularBufferSize_)
183  && configReader.lookupValue("mode", mode)) {
184  if(mode == "offline"){
185  mode_ = detail::RecoMode::offline;
186  }else if(mode == "online"){
187  mode_ = detail::RecoMode::online;
188  }else{

190  return EXIT_FAILURE;
191  }

193  }
