77 auto Backprojection::process(input_type&& sinogram) -> void {
78  if (sinogram.valid()) {
79  BOOST_LOG_TRIVIAL(debug)<< "BP: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
80  sinograms_[sinogram.device()].push(std::move(sinogram));
81  } else {
82  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Backprojection: Received sentinel, finishing.";
83 
84  //send sentinal to processor thread and wait 'til it's finished
85  for(auto i = 0; i < numberOfDevices_; i++) {
86  sinograms_[i].push(input_type());
87  }
88  for(auto i = 0; i < numberOfDevices_; i++) {
89  processorThreads_[i].join();
90  }
91 
92  results_.push(output_type());
93  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Backprojection: Finished.";
94  }
95 }

101 auto Backprojection::processor(const int deviceID) -> void {
102  CHECK(cudaSetDevice(deviceID));
103 
104  //init lookup tables for sin and cos
105  std::vector<float> sinLookup_h(numberOfProjections_), cosLookup_h(
106  numberOfProjections_);
107  for (auto i = 0; i < numberOfProjections_; i++) {
108  float theta = i * M_PI
109  / (float) numberOfProjections_+ rotationOffset_ / 180.0 * M_PI;
110  while (theta < 0.0) {
111  theta += 2.0 * M_PI;
112  }
113  sincosf(theta, &sinLookup_h[i], &cosLookup_h[i]);
114  }
115  CHECK(
116  cudaMemcpyToSymbol(sinLookup, sinLookup_h.data(),
117  sizeof(float) * numberOfProjections_));
118  CHECK(
119  cudaMemcpyToSymbol(cosLookup, cosLookup_h.data(),
120  sizeof(float) * numberOfProjections_));
121  //constants for kernel
122  const float scale_h = numberOfDetectors_ / (float) numberOfPixels_;
123  const float normalizationFactor_h = M_PI / numberOfProjections_ / scale_h;
124  const float imageCenter_h = (numberOfPixels_ - 1.0) * 0.5;
125  CHECK(cudaMemcpyToSymbol(normalizationFactor, &normalizationFactor_h, sizeof(float)));
126  CHECK(cudaMemcpyToSymbol(scale, &scale_h, sizeof(float)));
127  CHECK(cudaMemcpyToSymbol(imageCenter, &imageCenter_h, sizeof(float)));
128  dim3 blocks(blockSize2D_, blockSize2D_);
129  dim3 grids(std::ceil(numberOfPixels_ / (float) blockSize2D_),
130  std::ceil(numberOfPixels_ / (float) blockSize2D_));
131  if(interpolationType_ == detail::InterpolationType::linear)
132  CHECK(cudaFuncSetCacheConfig(backProjectLinear, cudaFuncCachePreferL1));
133  else if(interpolationType_ == detail::InterpolationType::neareastNeighbor)
134  CHECK(cudaFuncSetCacheConfig(backProjectNearest, cudaFuncCachePreferL1));
135  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::BP: Running Thread for Device " << deviceID;
136  while (true) {
137  //execution is blocked until next element arrives in queue
138  auto sinogram = sinograms_[deviceID].take();
139  //if sentinel, finish thread execution
140  if (!sinogram.valid())
141  break;
142 
143  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Backprojection: Backprojecting sinogram with Index " << sinogram.index();
144 
145  //allocate device memory for reconstructed picture
146  auto recoImage =
147  ddrf::MemoryPool<deviceManagerType>::instance()->requestMemory(
148  memoryPoolIdxs_[deviceID]);
149 
150  if(useTextureMemory_){
151  cudaResourceDesc resDesc;
152  memset(&resDesc, 0, sizeof(resDesc));
153  resDesc.resType = cudaResourceTypeLinear;
154  resDesc.res.linear.devPtr = sinogram.container().get();
155  resDesc.res.linear.desc.f = cudaChannelFormatKindFloat;
156  resDesc.res.linear.desc.x = 32; // bits per channel
157  resDesc.res.linear.sizeInBytes = sinogram.size()*sizeof(float);
158 
159  cudaTextureDesc texDesc;
160  memset(&texDesc, 0, sizeof(texDesc));
161  texDesc.addressMode[0] = cudaAddressModeBorder;
162  texDesc.addressMode[1] = cudaAddressModeBorder;
163  texDesc.normalizedCoords = false;
164 
165  // create texture object: we only have to do this once!
166  cudaTextureObject_t tex=0;
167  CHECK(cudaCreateTextureObject(&tex, &resDesc, &texDesc, NULL));
168  backProjectTex<<<grids, blocks, 0, streams_[deviceID]>>>(tex, recoImage.container().get(),
169  numberOfPixels_, numberOfProjections_, numberOfDetectors_);
170  CHECK(cudaDestroyTextureObject(tex));
171  }else{
172  if(interpolationType_ == detail::InterpolationType::linear)
173  backProjectLinear<<<grids, blocks, 0, streams_[deviceID]>>>(
174  sinogram.container().get(), recoImage.container().get(),
175  numberOfPixels_, numberOfProjections_, numberOfDetectors_);
176  else if(interpolationType_ == detail::InterpolationType::neareastNeighbor)
177  backProjectNearest<<<grids, blocks, 0, streams_[deviceID]>>>(
178  sinogram.container().get(), recoImage.container().get(),
179  numberOfPixels_, numberOfProjections_, numberOfDetectors_);
180  }
181  CHECK(cudaPeekAtLastError());
182 
183  recoImage.setIdx(sinogram.index());
184  recoImage.setDevice(deviceID);
185  recoImage.setPlane(sinogram.plane());
186  recoImage.setStart(sinogram.start());
187 
188  //wait until work on device is finished
189  CHECK(cudaStreamSynchronize(streams_[deviceID]));
190  results_.push(std::move(recoImage));
191 
192  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Backprojection: Reconstructing sinogram with Index " << sinogram.index() << " finished.";
193  }
194 }

224 __global__ void backProjectLinear(const float* const __restrict__ sinogram,
225  float* __restrict__ image,
226  const int numberOfPixels,
227  const int numberOfProjections,
228  const int numberOfDetectors){
229 
230  const auto x = ddrf::cuda::getX();
231  const auto y = ddrf::cuda::getY();
232 
233  float sum = 0.0;
234 
235  if(x >= numberOfPixels || y >= numberOfPixels)
236  return;
237 
238  const int centerIndex = numberOfDetectors * 0.5;
239 
240  const float xp = (x - imageCenter[0]) * scale[0];
241  const float yp = (y - imageCenter[0]) * scale[0];
242 
243 #pragma unroll 16
244  for(auto projectionInd = 0; projectionInd < numberOfProjections; projectionInd++){
245  const float t = xp * cosLookup[projectionInd] + yp * sinLookup[projectionInd];
246  const int a = floor(t);
247  const int aCenter = a + centerIndex;
248  if(aCenter >= 0 && aCenter < numberOfDetectors){
249  sum = sum + ((float)(a + 1) - t) * sinogram[projectionInd * numberOfDetectors + aCenter];
250  }
251  if((aCenter + 1) >= 0 && (aCenter + 1) < numberOfDetectors){
252  sum = sum + (t - (float)a) * sinogram[projectionInd * numberOfDetectors + aCenter + 1];
253  }
254 
255  }
256  image[x + y * numberOfPixels] = sum * normalizationFactor[0];
257 }

274

292  float* __restrict__ image, const int numberOfPixels,
293  const int numberOfProjections, const int numberOfDetectors) {
294 
295  const auto x = ddrf::cuda::getX();
296  const auto y = ddrf::cuda::getY();
297 
298  if (x >= numberOfPixels || y >= numberOfPixels)
299  return;
300 
301  float sum = 0.0;
302 
303  float *p_cosLookup = cosLookup;
304  float *p_sinLookup = sinLookup;
305 
306  const float scale = numberOfPixels / (float) numberOfDetectors;
307  const int centerIndex = numberOfDetectors * 0.5;
308 
309  const float xp = (x - (numberOfPixels - 1.0) * 0.5) / scale;
310  const float yp = (y - (numberOfPixels - 1.0) * 0.5) / scale;
311 
312 #pragma unroll 16
313  for (auto projectionInd = 0; projectionInd < numberOfProjections;
314  projectionInd++) {
315  //const int t = round(xp * cosLookup[projectionInd] + yp * sinLookup[projectionInd]) + centerIndex;
316  const int t = round(xp * *p_cosLookup + yp * *p_sinLookup) + centerIndex;
317  ++p_cosLookup; ++p_sinLookup;
318  if (t >= 0 && t < numberOfDetectors)
319  sum += sinogram[projectionInd * numberOfDetectors + t];
320  }
321  image[x + y * numberOfPixels] = sum * M_PI / numberOfProjections * scale;
322 }
