147 auto Fan2Para::process(input_type&& sinogram) -> void {
148  if (sinogram.valid()) {
149  BOOST_LOG_TRIVIAL(debug)<< "Fan2Para: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
150  fanSinograms_[sinogram.device()].push(std::move(sinogram));
151  } else {
152  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Fan2Para: Received sentinel, finishing.";
153 
154  //send sentinal to processor thread and wait 'til it's finished
155  for(auto i = 0; i < numberOfDevices_; i++) {
156  fanSinograms_[i].push(input_type());
157  }
158 
159  for(auto i = 0; i < numberOfDevices_; i++) {
160  processorThreads_[i].join();
161  }
162  //push sentinel to results for next stage
163  results_.push(output_type());
164  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Fan2Para: Finished.";
165  }
166 }

176 auto Fan2Para::processor(const int deviceID) -> void {
177  //nvtxNameOsThreadA(pthread_self(), "Fan2Para");
178  CHECK(cudaSetDevice(deviceID));
179  dim3 blocks2D(blockSize2D_, blockSize2D_);
180  dim3 grids2D(
181  std::ceil(params_.numberOfParallelDetectors_ / (float) blockSize2D_),
182  std::ceil(
183  params_.numberOfParallelProjections_ / (float) blockSize2D_));
184  dim3 blocks1D(blockSize1D_);
185  dim3 grids1D(
186  std::ceil(
187  params_.numberOfParallelProjections_
188  * params_.numberOfParallelDetectors_
189  / (float) blockSize1D_));
190  auto params_d = ddrf::cuda::make_device_ptr<parameters>(1);
191  CHECK(
192  cudaMemcpyAsync(params_d.get(), &params_, sizeof(parameters),

195  CHECK(cudaFuncSetCacheConfig(interpolation, cudaFuncCachePreferL1));
196  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Fan2Para: Running Thread for Device " << deviceID;
197  while (true) {
198  auto sinogram = fanSinograms_[deviceID].take();
199  if (!sinogram.valid())

203  //copy image from device to host

219  gammaBeforeRay1_d_[deviceID].get(),

222  gammaAfterRay2_d_[deviceID].get(), thetaGoalRay1_d_[deviceID].get(),

238 auto Fan2Para::computeFan2ParaTransp() -> void {
239 
240  BOOST_LOG_TRIVIAL(info)<< "Computing Hash Table for conversion from fan to parallel beam.";
241 
242  auto dataSetSize = params_.numberOfParallelProjections_ * params_.numberOfParallelDetectors_
243  * params_.numberOfPlanes_;
244 
245  //allocate memory on host
246  theta_.resize(params_.numberOfFanProjections_);
247  gamma_.resize(params_.numberOfFanDetectors_);
248  s_.resize(params_.numberOfParallelDetectors_);
249  alphaCircle_.resize(params_.numberOfParallelProjections_);
250 
251  thetaAfterRay1_.resize(dataSetSize);
252  thetaAfterRay2_.resize(dataSetSize);
253  thetaBeforeRay1_.resize(dataSetSize);
254  thetaBeforeRay2_.resize(dataSetSize);
255  gammaAfterRay1_.resize(dataSetSize);
256  gammaAfterRay2_.resize(dataSetSize);
257  gammaBeforeRay1_.resize(dataSetSize);
258  gammaBeforeRay2_.resize(dataSetSize);
259  gammaGoalRay1_.resize(dataSetSize);
260  gammaGoalRay2_.resize(dataSetSize);
261  thetaGoalRay1_.resize(dataSetSize);
262  thetaGoalRay2_.resize(dataSetSize);
263  ray1_.resize(dataSetSize);
264  ray2_.resize(dataSetSize);
265 
266  // === Init values for Hash table
267  // ===============================
268  // == Theta = Ortswinkel des Quellpunktes auf Target
269  // == Gamma = Ortswinkel des Detektorpixels
270  // == s = diskreter Abstand der Detektorpixel (Para)
271  // == alpha_kreis = Ortswinkel der Parallelstrahlquellen
272 
273  for (auto j = 0; j < params_.numberOfFanProjections_; j++) {
274  theta_[j] = j * (360.0 / params_.numberOfFanProjections_) - params_.sourceOffset_;
275  if (theta_[j] < 0.0)
276  theta_[j] = theta_[j] + 360.0;
277  theta_[j] = ((2 * M_PI) / 360.0) * theta_[j];
278  }
279 
280  gamma_[0] = 0.0;
281  for (auto j = 1; j < params_.numberOfFanDetectors_; j++) {
282  gamma_[j] = j * ((360.0 / (float) params_.numberOfFanDetectors_));
283  gamma_[j] = ((2 * M_PI) / 360.0) * gamma_[j];
284  }
285 
286  for (auto j = 0; j < params_.numberOfParallelDetectors_; j++)
287  s_[j] = ((-0.5) * params_.imageWidth_)
288  + (((0.5 + j) * params_.imageWidth_) / (float) params_.numberOfParallelDetectors_);
289 
290  for (auto j = params_.numberOfParallelProjections_; j > 0; j--) {
291  alphaCircle_[j] = j * (360.0 / (float) params_.numberOfParallelProjections_);
292  alphaCircle_[j] = ((2 * M_PI) / 360.0) * alphaCircle_[j] + M_PI / 2;
293 
294  if (alphaCircle_[j] > 2 * M_PI)
295  alphaCircle_[j] = (alphaCircle_[j]) - (2.0 * M_PI);
296  }
297 
298  // === calculate Hash Table
299  // =========================
300  unsigned long long ind = 0;
301  int i, j, k;
302  float kappa = 0, L = 0;
303  float tb = 0.0;
304 
305  // Abfrage
306  if (params_.imageCenterY_ != 0) {
307  L = sqrt(params_.imageCenterY_ * params_.imageCenterY_ + params_.imageCenterX_ * params_.imageCenterX_);
308 
309  if (params_.imageCenterY_ < 0)
310  tb = 1.0;
311 
312  kappa = atan(params_.imageCenterX_ / params_.imageCenterY_) + tb * M_PI;
313  } else if (params_.imageCenterX_ != 0) {
314  L = sqrt(params_.imageCenterY_ * params_.imageCenterY_ + params_.imageCenterX_ * params_.imageCenterX_);
315 
316  if (params_.imageCenterX_ < 0)
317  kappa = -M_PI / 2.;
318  else
319  kappa = M_PI / 2.;
320  }
321 
322  unsigned int parallelSize = params_.numberOfParallelDetectors_
323  * params_.numberOfParallelProjections_;
324  float temp_1;
325 
326  for (k = 0; k < params_.numberOfPlanes_; k++) {
327  for (j = 0; j < params_.numberOfParallelProjections_; j++) {
328  for (i = 0; i < params_.numberOfParallelDetectors_; i++) {
329 
330  ind = j * params_.numberOfParallelDetectors_ + i + (k * parallelSize);
331  temp_1 = (s_[i] - L * sin(alphaCircle_[j] - kappa)) / params_.rDetector_;
332 
333  //Prfen, ob asin mglich
334  if (temp_1 <= 1 || temp_1 >= -1)
335  computeAngles(i, j, ind, k, L, kappa);
336  }
337  }
338  }

341 auto Fan2Para::computeAngles(int i, int j, unsigned int ind, int k, float L,
342  float kappa) -> void {
343 
344  //bergabe-Parameter
345  float epsilon = 0;
346 
347  //Hilfsvariable
348  float dif_best = M_PI, dif = 0, best_x = 0, temp_1 = 0, temp_2 = 0;
349  int x = 0;
350 
351  //Berechnungsvorschrift
352  //Theta
353  temp_1 = asin(((s_[i] - L * sin(alphaCircle_[j] - kappa)) / rTarget_[k])); //<-----Vernderung
354  thetaGoalRay1_[ind] = alphaCircle_[j] - temp_1;
355 
356  if (thetaGoalRay1_[ind] < 0)
357  thetaGoalRay1_[ind] = thetaGoalRay1_[ind] + 2.0 * M_PI;
358 
359  thetaGoalRay1_[ind] = ellipse_kreis_uwe(thetaGoalRay1_[ind], deltaX_[k],
360  deltaZ_[k], 2 * rTarget_[k]);
361 
362  thetaGoalRay2_[ind] = alphaCircle_[j] + temp_1 - M_PI;
363  if (thetaGoalRay2_[ind] < 0)
364  thetaGoalRay2_[ind] = thetaGoalRay2_[ind] + 2.0 * M_PI;
365 
366  thetaGoalRay2_[ind] = ellipse_kreis_uwe(thetaGoalRay2_[ind], deltaX_[k],
367  deltaZ_[k], 2 * rTarget_[k]);
368 
369  temp_1 = ((360.0 - sourceAngle_[k]) / 2.0) / 180.0 * M_PI;
370  temp_2 = (360.0 - ((360.0 - sourceAngle_[k]) / 2.0)) / 180.0 * M_PI;
371  if (thetaGoalRay1_[ind] > temp_1 && thetaGoalRay1_[ind] < temp_2)
372  ray1_[ind] = 1;
373  if (thetaGoalRay2_[ind] > temp_1 && thetaGoalRay2_[ind] < temp_2)
374  ray2_[ind] = 1;
375 
376  epsilon = asin(
377  ((s_[i] - L * sin(alphaCircle_[j] - kappa)) / params_.rDetector_)); //<-----Vernderung
378 
379  if (ray1_[ind]) {
380 
381  //Gamma
382  gammaGoalRay1_[ind] = epsilon + alphaCircle_[j] - 1.5 * M_PI;
383  if (gammaGoalRay1_[ind] < 0)
384  gammaGoalRay1_[ind] = gammaGoalRay1_[ind] + 2.0 * M_PI;
385  if (gammaGoalRay1_[ind] > 2 * M_PI)
386  gammaGoalRay1_[ind] = gammaGoalRay1_[ind] - 2.0 * M_PI;
387 
388  //Vektor Teta nach Wert durchsuchen fr Fall 1
389  for (x = 0; x < params_.numberOfFanProjections_; x++) {
390  if (thetaGoalRay1_[ind] <= theta_[x]) {
391  dif = theta_[x] - thetaGoalRay1_[ind];
392  if (dif < dif_best) {
393  dif_best = dif;
394  best_x = x;
395  }
396  }
397  }
398 
399  if (best_x == 0) {
400  thetaBeforeRay1_[ind] = params_.numberOfFanProjections_ - 1;
401  thetaAfterRay1_[ind] = best_x;
402  } else {
403  thetaBeforeRay1_[ind] = best_x - 1;
404  thetaAfterRay1_[ind] = best_x;
405  }
406 
407  //Vektor Gamma nach Wert durchsuchen fr Fall 1
408  for (x = 0; x < params_.numberOfFanDetectors_; x++) {
409  if (gammaGoalRay1_[ind] <= gamma_[x]) {
410  if (x == 0)
411  gammaBeforeRay1_[ind] = params_.numberOfFanDetectors_ - 1;
412  else
413  gammaBeforeRay1_[ind] = x - 1;
414  gammaAfterRay1_[ind] = x;
415  break;
416  }
417  }
418  if (gammaGoalRay1_[ind] > gamma_[params_.numberOfFanDetectors_ - 1]) {
419  gammaBeforeRay1_[ind] = params_.numberOfFanDetectors_ - 1;
420  gammaAfterRay1_[ind] = 0;
421  }
422  }
423 
424  if (ray2_[ind]) {
425  dif_best = M_PI;
426 
427  //Gamma fr Fall 2
428  gammaGoalRay2_[ind] = -epsilon + alphaCircle_[j] - (M_PI / 2.0);
429  if (gammaGoalRay2_[ind] < 0)
430  gammaGoalRay2_[ind] = gammaGoalRay2_[ind] + 2.0 * M_PI;
431 
432  //Vektor Teta nach Wert durchsuchen fr Fall 2
433  for (x = 0; x < params_.numberOfFanProjections_; x++) {
434  if (thetaGoalRay2_[ind] <= theta_[x]) {
435  dif = theta_[x] - thetaGoalRay2_[ind];
436  if (dif < dif_best) {
437  dif_best = dif;
438  best_x = x;
439  }
440  }
441  }
442  if (best_x == 0) {
443  thetaBeforeRay2_[ind] = params_.numberOfFanProjections_ - 1;
444  thetaAfterRay2_[ind] = best_x;
445 
446  } else {
447  thetaBeforeRay2_[ind] = best_x - 1;
448  thetaAfterRay2_[ind] = best_x;
449  }
450 
451  //Vektor Gamma nach Wert durchsuchen fr Fall 2
452  for (x = 0; x < params_.numberOfFanDetectors_; x++) {
453  if (gammaGoalRay2_[ind] <= gamma_[x]) {
454  if (x == 0)
455  gammaBeforeRay2_[ind] = params_.numberOfFanDetectors_ - 1;
456  else
457  gammaBeforeRay2_[ind] = x - 1;
458  gammaAfterRay2_[ind] = x;
459  break;
460  }
461  }
462  if (gammaGoalRay2_[ind] > gamma_[params_.numberOfFanDetectors_ - 1]) {
463  gammaBeforeRay2_[ind] = params_.numberOfFanDetectors_ - 1;
464  gammaAfterRay2_[ind] = 0;
465  }
466  }
