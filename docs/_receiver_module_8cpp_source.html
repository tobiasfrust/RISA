72  //Possibility how to realize timeout with boost::asio::udp was not found yet

106 auto ReceiverModule::readConfig(const std::string& configFile) -> bool {
107  ConfigReader configReader = ConfigReader(configFile.data());
108  int samplingRate, scanRate;
109  std::string transportProt;
110  if (configReader.lookupValue("samplingRate", samplingRate)
111  && configReader.lookupValue("numberOfFanDetectors", numberOfDetectors_)
112  && configReader.lookupValue("scanRate", scanRate)
113  && configReader.lookupValue("transportProtocol", transportProt)
114  && configReader.lookupValue("timeout", timeout_)
115  && configReader.lookupValue("numberOfDetectorsPerModule", numberOfDetectorsPerModule_)
116  && configReader.lookupValue("numberOfProjectionsPerPacket", numberOfProjectionsPerPacket_)
117  && configReader.lookupValue("inputBufferSize", bufferSize_)
118  && configReader.lookupValue("numberOfDetectorModules", numberOfDetectorModules_)) {
119  if(transportProt == "udp")
120  transportProtocol_ = transportProtocol::UDP;
121  else if(transportProt == "tcp")
122  transportProtocol_ = transportProtocol::TCP;
123  else
124  transportProtocol_ = transportProtocol::UDP;
125  numberOfProjections_ = samplingRate * 1000000 / scanRate;
126  return EXIT_SUCCESS;
127  }
128 
129  return EXIT_FAILURE;
130 }
