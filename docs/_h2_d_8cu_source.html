71 auto H2D::process(input_type&& sinogram) -> void {
72  if (sinogram.valid()) {
73  if(sinogram.index() > 0)
74  tmr_.stop();
75  BOOST_LOG_TRIVIAL(debug) << "H2D: Image arrived with Index: " << sinogram.index() << "to device " << lastDevice_;
76 // int device = sinogram.index() % 5;
77 // if(device == 0) device = 1;
78 // else device = 0;
79  sinograms_[lastDevice_].push(std::move(sinogram));
80  lastDevice_ = (lastDevice_ + 1) % numberOfDevices_;
81  double time = tmr_.elapsed();
82  if(sinogram.index() > 0){
83  if(time < bestCaseTime_)
84  bestCaseTime_ = time;
85  if(time > worstCaseTime_)
86  worstCaseTime_ = time;
87  }
88  tmr_.start();
89  int diff = sinogram.index() - lastIndex_ - 1;
90  lostSinos_ += diff;
91  if(diff > 0)
92  BOOST_LOG_TRIVIAL(debug) << "Skipping " << diff << " elements.";
93  if(count_%10000 == 0)
94  BOOST_LOG_TRIVIAL(info) << "Did not process " << lostSinos_ << " elements; " << lostSinos_/(double)lastIndex_*100.0 << "% loss";
95  count_++;
96  lastIndex_ = sinogram.index();
97  } else {
98  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::H2D: Received sentinel, finishing.";
99 
100  //send sentinel to all processor threads and wait 'til they're finished
101  for(auto i = 0; i < numberOfDevices_; i++){
102  sinograms_[i].push(input_type());
103  }
104 
105  //wait until all threads are finished
106  for(auto i = 0; i < numberOfDevices_; i++){
107  processorThreads_[i].join();
108  }
109 
110  //push sentinel to results for next stage
111  results_.push(output_type());
112 
113  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::H2D: Finished.";
114  }
115 }

121 auto H2D::processor(int deviceID) -> void {
122  //nvtxNameOsThreadA(pthread_self(), "H2D");
123  CHECK(cudaSetDevice(deviceID));
124  //for conversion from short to float
125  std::vector<float> temp(numberOfProjections_*numberOfDetectors_);
126  auto inputShort_d = ddrf::cuda::make_device_ptr<unsigned short>(numberOfProjections_*numberOfDetectors_);
127  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::H2D: Running Thread for Device " << deviceID;
128  while (true) {
129  auto sinogram = sinograms_[deviceID].take();
130  if (!sinogram.valid())
131  break;
132 
133  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::H2D: Copy sinogram " << sinogram.index() << " to device " << deviceID;
134 
135  //copy image from device to host
136  auto img = ddrf::MemoryPool<deviceManagerType>::instance()->requestMemory(
137  memoryPoolIdxs_[deviceID]);
138 
139  CHECK(
140  cudaMemcpyAsync(img.container().get(),sinogram.container().get(),
141  sinogram.size() * sizeof(unsigned short), cudaMemcpyHostToDevice, streams_[deviceID]));
142 
143  //needs to be set due to reuse of memory
144  img.setIdx(sinogram.index());
145  img.setDevice(deviceID);
146  img.setPlane(sinogram.plane());
147  img.setStart(sinogram.start());
148 
149  CHECK(cudaStreamSynchronize(streams_[deviceID]));
150 
151  //wait until work on device is finished
152  results_.push(std::move(img));
153 
154  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::H2D: Copy sinogram " << sinogram.index() << " to device finished.";
155  }
