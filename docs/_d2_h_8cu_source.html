70 auto D2H::process(input_type&& img) -> void {
71  if (img.valid()) {
72  if(img.index() == 0)
73  tmr_.start();
74  if((count_ % 10000) == 9999){
75  tmr_.stop();
76  reconstructionRate_ = (reconstructionRate_*(counter_-1.0) + 10000.0/(tmr_.elapsed())) / counter_;
77  counter_ += 1.0;
78  BOOST_LOG_TRIVIAL(info) << "Reconstructing at " << 10000.0/(tmr_.elapsed()) << " Images/second.";
79  tmr_.start();
80  }
81  count_++;
82  BOOST_LOG_TRIVIAL(debug)<< "Image " << img.index() << "from device " << img.device() << "arrived";
83  imgs_[img.device()].push(std::move(img));
84  } else {
85  BOOST_LOG_TRIVIAL(debug)<< "cuda::D2H: Received sentinel, finishing.";
86 
87  //send sentinal to processor threads and wait 'til they're finished
88  for(auto i = 0; i < numberOfDevices_; i++) {
89  imgs_[i].push(input_type());
90  }
91 
92  for(auto i = 0; i < numberOfDevices_; i++) {
93  processorThreads_[i].join();
94  }
95 
96  //push sentinel to results for next stage
97  results_.push(output_type());
98  BOOST_LOG_TRIVIAL(info) << "cuda::D2H: Finished.";
99  }
100 }

106 auto D2H::processor(const int deviceID) -> void {
107  //nvtxNameOsThreadA(pthread_self(), "D2H");
108  CHECK(cudaSetDevice(deviceID));
109  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::D2H: Running Thread for Device " << deviceID;
110  while (true) {
111  auto img = imgs_[deviceID].take();
112  if (!img.valid()) {
113  BOOST_LOG_TRIVIAL(debug)<< "invalid image arrived.";
114  break;
115  }
116 
117  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::D2H: Copy sinogram " << img.index() << " from device " << img.device();
118 
119  //copy image from device to host
120  auto ret = ddrf::MemoryPool<hostManagerType>::instance()->requestMemory(
121  memoryPoolIdx_);
122  CHECK(
123  cudaMemcpyAsync(ret.container().get(), img.container().get(),
124  img.size() * sizeof(float), cudaMemcpyDeviceToHost, streams_[deviceID]));
125  ret.setIdx(img.index());
126  ret.setPlane(img.plane());
127  ret.setStart(img.start());
128  CHECK(cudaStreamSynchronize(streams_[deviceID]));
129 
130  //wait until work on device is finished
131  results_.push(std::move(ret));
132 
133  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::D2H: Copy sinogram " << img.index() << " from device " << img.device() << " finished.";
134  }
135 }
